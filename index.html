function makeMove(fromR, fromC, toR, toC, shouldSend = true) { if (gameOver) return; // Находим исходную ячейку и фигуру const fromCell = document.querySelector(`[data-row="${fromR}"][data-col="${fromC}"]`); const piece = fromCell?.querySelector('.piece'); if (!piece || !fromCell) { // fallback board[toR][toC] = board[fromR][fromC]; board[fromR][fromC] = 0; performCaptures(toR, toC); checkWinConditions(); currentPlayer = currentPlayer === DEF ? ATT : DEF; if (shouldSend && conn?.open) conn.send({type: 'move', fromR, fromC, toR, toC}); render(); return; } // Создаём клон для анимации const clone = piece.cloneNode(true); clone.style.position = 'absolute'; clone.style.zIndex = '50'; clone.style.pointerEvents = 'none'; clone.style.transition = 'transform 0.6s cubic-bezier(0.25, 0.1, 0.25, 1), opacity 0.6s ease'; // Позиционируем клон точно над исходной фигурой const boardRect = document.getElementById('board').getBoundingClientRect(); const fromRect = fromCell.getBoundingClientRect(); clone.style.left = (fromRect.left - boardRect.left) + 'px'; clone.style.top = (fromRect.top - boardRect.top) + 'px'; clone.style.width = fromRect.width + 'px'; clone.style.height = fromRect.height + 'px'; document.getElementById('board').appendChild(clone); // Скрываем оригинал, чтобы не двоилось piece.style.opacity = '0'; // Находим целевую ячейку const toCell = document.querySelector(`[data-row="${toR}"][data-col="${toC}"]`); const toRect = toCell.getBoundingClientRect(); const dx = toRect.left - fromRect.left; const dy = toRect.top - fromRect.top; // Запускаем анимацию (requestAnimationFrame помогает увидеть начальное состояние) requestAnimationFrame(() => { requestAnimationFrame(() => { // двойной raf — часто решает проблему "анимация не стартует" clone.style.transform = `translate(${dx}px, ${dy}px)`; clone.style.opacity = '1'; }); }); // Когда анимация закончится const finishMove = () => { clone.remove(); piece.style.opacity = ''; board[toR][toC] = board[fromR][fromC]; board[fromR][fromC] = 0; performCaptures(toR, toC); checkWinConditions(); currentPlayer = currentPlayer === DEF ? ATT : DEF; if (shouldSend && conn?.open) { conn.send({type: 'move', fromR, fromC, toR, toC}); } render(); }; clone.addEventListener('transitionend', finishMove, {once: true}); // Страховка на 900 мс setTimeout(() => { if (clone.parentNode) finishMove(); }, 900); }
